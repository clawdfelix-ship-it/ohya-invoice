function sendInvoiceEmail(id) { var inv = invoices.find(function(i){ return i.id === id; }); if (!inv) return; var email = inv.client_email; if (!email) { alert('這個客戶沒有電郵地址，請先在客戶資料中填寫電郵。'); return; } var total = (inv.items || []).reduce(function(s, it) { return s + (it.qty || 1) * (it.price || 0); }, 0); var subject = '發票 ' + (inv.no || '') + ' - ' + (inv.client || ''); var body = '您好,\n\n請查閱附件中的發票\n\n發票編號：' + (inv.no || '') + '\n日期：' + (inv.date || '') + '\n金額：$' + total.toFixed(2) + '\n\n謝謝'; window.open('mailto:' + encodeURIComponent(email) + '?subject=' + encodeURIComponent(subject) + '&body=' + encodeURIComponent(body), '_blank'); } // 頁面初始化：載入數據 document.addEventListener('DOMContentLoaded', async function() { try { await loadData(); renderDashboard(); renderInvoices(); renderCustomers(); renderProducts(); renderRecords(); } catch (err) { console.error('Failed to load data:', err); } });
